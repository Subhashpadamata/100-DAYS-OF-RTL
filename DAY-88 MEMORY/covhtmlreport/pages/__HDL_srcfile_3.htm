1  module memory(clk_i,rst_i,addr_i,wr_rd_i,wdata_i,valid_i,rdata_o,ready_o);
     2  parameter WIDTH=16;
     3  parameter DEPTH=512;
     4  parameter ADDR_WIDTH=$clog2(DEPTH);
     5  input clk_i,rst_i,valid_i,wr_rd_i;
     6  input [ADDR_WIDTH-1:0]addr_i;
     7  input [WIDTH-1:0]wdata_i;
     8  output reg [WIDTH-1:0]rdata_o;
     9  output reg ready_o;
     10  integer i;
     11  reg [WIDTH-1:0] mem[DEPTH-1:0];
     12  	
     13  	always@(posedge clk_i)begin
     14  		if(rst_i)begin
     15  			ready_o=0;
     16  			rdata_o=0;
     17  
     18  			for(i=0;i<WIDTH;i=i+1) begin
     19  				mem[i]=0;
     20  			end
     21  		end
     22  		else begin
     23  			if(valid_i)begin
     24  				ready_o=1;
     25  					if(wr_rd_i)begin
     26  						mem[addr_i]=wdata_i;
     27  					end
     28  					else begin
     29  						rdata_o=mem[addr_i];
     30  					end
     31  			end
     32  			else begin
     33  				ready_o=0;
     34  			end
     35  		end
     36  	end
     37  endmodule
